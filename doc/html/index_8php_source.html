670 AUTHOR : udo dot schroeter at gmail dot com 26-May-2007 06:40
671 SCRIPT URL : http://php.net/manual/fr/function.eval.php
672 COMMENTS :
673 Safer Eval
674 
675 eval() is used way to often. It slows down code, makes it harder to maintain and it created security risks. However, sometimes, I found myself wishing I could allow some user-controlled scripting in my software, without giving access to dangerous functions.
676 
677 That's what the following class does: it uses PHP's tokenizer to parse a script, compares every function call against a list of allowed functions. Only if the script is "clean", it gets eval'd.
678 
679 Usage example:
680 <?php
681  $ls = new SaferScript('horribleCode();');
682 
683  $ls->allowHarmlessCalls();
684  print_r($ls->parse());
685  $ls->execute();
686 */
687 class SaferScript {
688  var $source, $allowedCalls, $data;
689 
690  function SaferScript($scriptText, $data='') {
691  $this->source = $scriptText;
692  $this->allowedCalls = array();
693  $this->data = $data;
694  }

696  function allowHarmlessCalls() {
697  $this->allowedCalls = explode(',',
698  'explode,implode,date,time,round,rand,ceil,floor,srand,'.
699  'strtolower,strtoupper,substr,stristr,strpos,print,print_r,'.
700  'foreach,return,break,substr_replace,str_pad,intval,'.
701  'intval,floatval,strval,true,false,STR_PAD_LEFT,STR_PAD_RIGHT,STR_PAD_BOTH,bin2hex');
702  }

704  function parse() {
705  $this->parseErrors = array();
706  $tokens = token_get_all('<?'.'php '.$this->source.' ?'.'>');
707  $vcall = '';
708 
709  foreach ($tokens as $token) {
710  if (is_array($token)) {
711  $id = $token[0];
712  switch ($id) {
713  case(T_VARIABLE): { $vcall .= 'v'; break; }
714  case(T_STRING): { $vcall .= 's'; }
715  case(T_REQUIRE_ONCE): case(T_REQUIRE): case(T_NEW): case(T_RETURN):
716  case(T_BREAK): case(T_CATCH): case(T_CLONE): case(T_EXIT):
717  case(T_PRINT): case(T_GLOBAL): case(T_ECHO): case(T_INCLUDE_ONCE):
718  case(T_INCLUDE): case(T_EVAL): case(T_FUNCTION): {
719  if (array_search($token[1], $this->allowedCalls) === false)
720  $this->parseErrors[] = _("illegal call").': '.$token[1];
721  }
722  }
723  }
724  else
725  $vcall .= $token;
726  }
727 
728  if (stristr($vcall, 'v(') != '')
729  $this->parseErrors[] = array(_("illegal dynamic function call"));
730 
731  return($this->parseErrors);
732  }

735  foreach ($parameters as $k => $v)
736  $$k = $v;
737 
738  $data = $this->data;
739  if (get_magic_quotes_gpc()) {
740  $data=stripslashes($data);
741  } else {
742  $data=$data;
743  }
744 
745  if (sizeof($this->parseErrors) == 0)
746  return eval($this->source);
747  else
748  return false;//print('cannot execute, script contains errors');
749  }
